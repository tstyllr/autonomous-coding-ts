<project_specification>
  <project_name>Todo App - Task Management</project_name>

  <overview>
    Build a simple, functional Todo application with user authentication. Users can register, login,
    and manage their personal todo lists. Each user has their own isolated todo items with basic CRUD
    operations, status toggling, and filtering. The focus is on simplicity, stability, and using mature,
    well-tested technologies. This project is intended for testing purposes only.
  </overview>

  <technology_stack>
<framework>Next.js (App Router) - full-stack React framework</framework>
    <language>TypeScript</language>
    <database>SQLite with Prisma ORM</database>
    <authentication>NextAuth.js (credentials provider with email/password)</authentication>
    <styling>Tailwind CSS</styling>
    <password_hashing>bcrypt</password_hashing>
    <validation>zod</validation>
    <port>Only launch on port 3000</port>
  </technology_stack>

  <prerequisites>
    <environment_setup>
      - Node.js 18+ installed
      - pnpm as package manager
      - DATABASE_URL configured in .env (e.g. file:./dev.db)
      - NEXTAUTH_SECRET configured in .env
    </environment_setup>
  </prerequisites>

  <core_features>
    <authentication>
      - User registration with email and password
      - User login with email and password
      - User logout
      - Session management via NextAuth.js
      - Protected routes (redirect to login if not authenticated)
      - Password hashing with bcrypt
    </authentication>

    <todo_management>
      - Create a new todo item (title required, description optional)
      - List all todos for the authenticated user
      - Toggle todo completion status
      - Edit todo title and description
      - Delete a todo item
      - Filter todos by status (all, active, completed)
      - Display todo count (total and remaining)
      - Sort by creation date (newest first)
    </todo_management>

    <ui_design>
      - Clean, minimal interface
      - Responsive layout (mobile and desktop)
      - Light/dark mode support via Tailwind
      - Loading states for async operations
      - Error messages for form validation
      - Confirmation dialog for delete actions
    </ui_design>
  </core_features>

  <database_schema>
    <tables>
      <users>
        - id: Int (auto-increment, primary key)
        - email: String (unique)
        - password: String (hashed)
        - name: String (optional)
        - createdAt: DateTime (default now)
        - updatedAt: DateTime (auto-update)
      </users>

      <todos>
        - id: Int (auto-increment, primary key)
        - title: String
        - description: String (optional)
        - completed: Boolean (default false)
        - userId: Int (foreign key -> users.id)
        - createdAt: DateTime (default now)
        - updatedAt: DateTime (auto-update)
      </todos>
    </tables>
  </database_schema>

  <api_endpoints_summary>
    <authentication>
      - POST /api/auth/register - Register new user
      - POST /api/auth/[...nextauth] - NextAuth.js handlers (login, logout, session)
    </authentication>

    <todos>
      - GET /api/todos - List all todos for authenticated user (supports ?filter=all|active|completed)
      - POST /api/todos - Create a new todo
      - PUT /api/todos/:id - Update a todo (title, description, completed)
      - DELETE /api/todos/:id - Delete a todo
    </todos>
  </api_endpoints_summary>

  <ui_layout>
    <pages>
      <login_page>
        - Email and password input fields
        - Login button
        - Link to registration page
        - Error message display
      </login_page>

      <register_page>
        - Name, email, password, confirm password fields
        - Register button
        - Link to login page
        - Validation error messages
      </register_page>

      <home_page>
        - Header with app title, user name, and logout button
        - Add todo input field with submit button
        - Filter tabs: All / Active / Completed
        - Todo list with each item showing:
          - Checkbox to toggle completion
          - Title (strikethrough when completed)
          - Edit and delete buttons
          - Description (expandable)
        - Todo count summary at bottom
        - Empty state message when no todos
      </home_page>
    </pages>
  </ui_layout>

  <design_system>
    <color_palette>
      - Primary: Blue (#3B82F6)
      - Background: White (light), Dark gray (#111827 dark)
      - Surface: Gray-50 (light), Gray-800 (dark)
      - Text: Gray-900 (light), Gray-100 (dark)
      - Border: Gray-200 (light), Gray-700 (dark)
      - Success: Green-500 (completed items)
      - Danger: Red-500 (delete actions)
    </color_palette>

    <typography>
      - System font stack (Inter, system-ui, sans-serif)
      - Headings: font-bold
      - Body: font-normal
    </typography>
  </design_system>

      </tasks>

    <step number="3">
      <title>Todo API</title>
      <tasks>

        - Create POST /api/todos with input validation
        - Create PUT /api/todos/:id with ownership check
        - Create DELETE /api/todos/:id with ownership check
        - Add proper error handling and status codes
      </tasks>
    </step>

    <step number="4">
      <title>Todo UI</title>
      <tasks>

  <success_criteria>

    <technical_quality>
